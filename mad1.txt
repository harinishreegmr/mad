# 🧠 5. MainActivity.kt

```kotlin
class MainActivity : AppCompatActivity() {

    lateinit var progressBar: ProgressBar
    lateinit var recyclerView: RecyclerView
    lateinit var mapView: MapView
    lateinit var sharedPref: SharedPreferences

    override fun onCreate(savedInstanceState: Bundle?) {
        super.onCreate(savedInstanceState)
        setContentView(R.layout.activity_main)

        progressBar = findViewById(R.id.progressBar)
        recyclerView = findViewById(R.id.recyclerView)
        mapView = findViewById(R.id.mapView)

        sharedPref = getSharedPreferences("FoodApp", MODE_PRIVATE)

        setupMenu()
        showProgress()
        showNotification()
        saveUser()
        initMap()

        findViewById<Button>(R.id.dateBtn).setOnClickListener {
            Utils.showDatePicker(this)
        }
    }

    private fun setupMenu() {
        val list = listOf("Pizza", "Burger", "Biryani")
        recyclerView.layoutManager = LinearLayoutManager(this)
        recyclerView.adapter = FoodAdapter(list)
    }

    private fun showProgress() {
        progressBar.visibility = View.VISIBLE
        Handler(Looper.getMainLooper()).postDelayed({
            progressBar.visibility = View.GONE
        }, 2000)
    }

    private fun saveUser() {
        sharedPref.edit().putString("user", "Harinishree").apply()
    }

    private fun showNotification() {
        val manager = getSystemService(Context.NOTIFICATION_SERVICE) as NotificationManager

        val channel = NotificationChannel("food", "FoodApp", NotificationManager.IMPORTANCE_DEFAULT)
        manager.createNotificationChannel(channel)

        val notification = NotificationCompat.Builder(this, "food")
            .setContentTitle("Order Placed")
            .setContentText("Your food is on the way!")
            .setSmallIcon(android.R.drawable.ic_dialog_info)
            .build()

        manager.notify(1, notification)
    }

    private fun initMap() {
        mapView.getMapboxMap().loadStyleUri(Style.MAPBOX_STREETS)
    }
}
```

# 📦 7. FoodAdapter.kt

```kotlin
class FoodAdapter(private val list: List<String>) :
    RecyclerView.Adapter<FoodAdapter.ViewHolder>() {

    class ViewHolder(view: View) : RecyclerView.ViewHolder(view) {
        val text: TextView = view.findViewById(R.id.foodName)
    }

    override fun onCreateViewHolder(parent: ViewGroup, viewType: Int): ViewHolder {
        val view = LayoutInflater.from(parent.context)
            .inflate(R.layout.item_food, parent, false)
        return ViewHolder(view)
    }

    override fun onBindViewHolder(holder: ViewHolder, position: Int) {
        holder.text.text = list[position]
    }

    override fun getItemCount() = list.size
}
```

```kotlin
class SMSReceiver : BroadcastReceiver() {
    override fun onReceive(context: Context, intent: Intent) {
        val bundle = intent.extras
        val pdus = bundle?.get("pdus") as Array<*>

        for (pdu in pdus) {
            val msg = SmsMessage.createFromPdu(pdu as ByteArray)
            Log.d("SMS", msg.messageBody)
        }
    }
}
```

# 🧰 10. Utils.kt

```kotlin
object Utils {

    fun showDatePicker(context: Context) {
        val cal = Calendar.getInstance()
        DatePickerDialog(context,
            { _, y, m, d ->
                Toast.makeText(context, "$d/$m/$y", Toast.LENGTH_SHORT).show()
            },
            cal.get(Calendar.YEAR),
            cal.get(Calendar.MONTH),
            cal.get(Calendar.DAY_OF_MONTH)
        ).show()
    }

    fun getAddress(context: Context, lat: Double, lng: Double): String {
        val geocoder = Geocoder(context, Locale.getDefault())
        val address = geocoder.getFromLocation(lat, lng, 1)
        return address?.get(0)?.getAddressLine(0) ?: "Unknown"
    }

    fun checkWifi(context: Context): Boolean {
        val wifi = context.getSystemService(Context.WIFI_SERVICE) as WifiManager
        return wifi.isWifiEnabled
    }

    fun checkBluetooth(): Boolean {
        val bt = BluetoothAdapter.getDefaultAdapter()
        return bt?.isEnabled == true
    }

    fun playSound(context: Context) {
        val mediaPlayer = MediaPlayer.create(context, R.raw.notification)
        mediaPlayer.start()
    }
}
```
